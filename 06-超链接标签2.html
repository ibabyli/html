				public int add()
				{
					return num1+num2;
				}
				
				public int sub()
				{
					return num1-num2;
				}
				
				public int mul()
				{
					return num1*num2;
				}
				
				public int div()
				{
					return num1/num2;
				}
				
				public int mod()
				{
					return num1%num2;
				}
				
				public void printInfo()
				{
					System.out.println("加法：");
